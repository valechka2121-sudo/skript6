\documentclass[aspectratio=169]{beamer} \usetheme{default} \usecolortheme{dolphin} \setbeamerfont{title}{series=\bfseries} \setbeamerfont{frametitle}{series=\bfseries} \title{Пошаговая система первого звонка} \subtitle{Звонок, ведущий на очную консультацию} \author{} \date{} \begin{document} \begin{frame} \titlepage \end{frame} \begin{frame}{Философия и подготовка} \begin{itemize} \item \textbf{Тон голоса:} Спокойный, уверенный, сочувствующий. Никакой излишней веселости. \item \textbf{Цель звонка:} Не продать услугу «здесь и сейчас». Цель — выявить проблему, дать надежду и записать на очную консультацию к юристу. \end{itemize} \end{frame} \begin{frame}{Самоощущение} Ты — доктор скорой помощи. Человек звонит и говорит: «У меня болит». Твоя задача — поставить предварительный диагноз и сказать: «Приезжайте в клинику, мы вас осмотрим и вылечим». \end{frame} \begin{frame}{Подготовка к звонку} \textbf{Изучи базу:} Перед звонком посмотри, откуда лид (заявка с сайта, реклама в соцсетях). Это даст контекст. \end{frame} \begin{frame}{Этап 1. Установление контакта} \textbf{Задача:} Снять защитную реакцию, представиться, обозначить причину звонка. \vspace{1em} «Здравствуйте, [Имя Клиента]? Меня зовут [Твое Имя], я руководитель направления по банкротству физических лиц в компании [Название компании]. Вы недавно оставляли заявку/просили перезвонить по вопросу решения долговой ситуации. Я вам дозвонился?» \end{frame} \begin{frame}{Почему это работает} \begin{itemize} \item «Руководитель направления» — добавляет вес и экспертизу. \item «По вопросу решения долговой ситуации» — говорим о решении, а не о проблеме. \item «Я вам дозвонился?» — дает клиенту контроль над разговором. \end{itemize} \end{frame} \begin{frame}{Этап 2. Диагностика} «Чтобы я мог вам максимально точно помочь и сориентировать по возможным решениям, задам несколько коротких вопросов. Это займет буквально 2-3 минуты. Согласны?» \end{frame} \begin{frame}{Ключевые вопросы} \begin{itemize} \item «Подскажите, общая сумма ваших долгов, включая все кредиты, займы, штрафы, коммуналку, находится в каком диапазоне: до 300 тысяч, от 300 тыс. до 1 млн, или больше миллиона?» \item «Долги в основном перед банками или перед микрофинансовыми организациями (МФО)? Есть ли долги по ЖКХ или перед частными лицами?» \item «Сейчас платежи по этим долгам вы вносите регулярно, с задержками или уже давно не платите?» \item «Есть ли у вас в собственности какое-либо ценное имущество: квартира, машина, дача, гараж?» \end{itemize} \end{frame} \begin{frame}{Этап 3. Дать надежду} «[Имя Клиента], спасибо, что подробно рассказали. Исходя из того, что вы сказали (сумма долга большая/платежи вносить не получается), ваша ситуация — типовая для процедуры банкротства физического лица...» \end{frame} \begin{frame}{Суть решения} \begin{itemize} \item Процедура длится в среднем 7–10 месяцев \item Долги списываются законно и окончательно \item Вы получаете решение суда \item Единственное жилье сохраняется \end{itemize} \end{frame} \begin{frame}{Этап 4. Приглашение на консультацию} «Следующий шаг — это бесплатная очная консультация в нашем офисе... Скажите, в какой день на следующей неделе вам было бы удобно приехать: в среду или в четверг?» \end{frame} \begin{frame}{Этап 5. Возражения} \textbf{Это дорого / У меня нет денег} «На консультации специалист расскажет обо всех вариантах работы, включая рассрочку...» \end{frame} \begin{frame}{Возражение: Я подумаю} «Консультация ни к чему не обязывает, но даст вам полную ясность...» \end{frame} \begin{frame}{Возражение: Единственное жилье} «Единственное пригодное для проживания жилье по закону не отбирается...» \end{frame} \begin{frame}{Этап 6. Завершение звонка} «Отлично, [Имя Клиента]! Я вас записал на [Дата], [Время]... До встречи!» \end{frame} \begin{frame}{После звонка} В течение 5 минут отправь сообщение с напоминанием о записи, дате, времени и адресе. \end{frame} \begin{frame}{Главные правила} \begin{itemize} \item Слушай больше, чем говоришь \item Не давай юридических консультаций по телефону \item Будь на стороне клиента \item Работай со страхами \item Верь в то, что говоришь \end{itemize} \end{frame} \end{document}
